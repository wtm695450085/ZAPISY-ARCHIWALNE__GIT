epochs = 2500
aggregated_losses = []


learning_rate = 0.0001     #<= wielko krokw
for t in range(epochs):         #<= ilo epok
 
  h = X.mm(w1)               #<= zwyke mnoenie macierzy x*w1
  h_relu = h.clamp(min=0)    #<= wyznaczenie ograniczenia do min=0
  y_pred = h_relu.mm(w2)     #<= pomnoenie przez daje predykcj y_pred
## Wyznaczenie straty (loss) jako wskanika r2
## Strata Loss jest skalarem i jest przechowywana w tensorze PyTorcha(); 
## moemy uzyska jego warto jako liczb w jzyku Python za pomoc loss.item ().
  loss = (y_pred - y).pow(2).sum()
  
  aggregated_losses.append(loss) ## potrzebne do wykresu

  print('Krok:',t, loss.item())        # Dla kadej pentli drukuje wynik r2
  # Backprop do obliczania gradientw w1 i w2 w odniesieniu do strat
  grad_y_pred = 2.0 * (y_pred - y)      #bd modelu razy 2
  grad_w2 = h_relu.t().mm(grad_y_pred)  #
  grad_h_relu = grad_y_pred.mm(w2.t())
  grad_h = grad_h_relu.clone()
  grad_h[h < 0] = 0
  grad_w1 = X.t().mm(grad_h)

  # Zaktualizuj wagi przy uyciu spadku gradientu
  w1 -= learning_rate * grad_w1
  w2 -= learning_rate * grad_w2
